2. clusters/dev/helmrelease.yaml â€“ HelmRelease for Dev
apiVersion: helm.toolkit.fluxcd.io/v2beta1
kind: HelmRelease
metadata:
  name: myapp
  namespace: dev
spec:
  releaseName: myapp-dev
  interval: 5m
  chart:
    spec:
      chart: myapp
      version: "~1.2.0"
      sourceRef:
        kind: HelmRepository
        name: myapp-helm
        namespace: flux-system
  valuesFrom:
    - kind: ConfigMap
      name: myapp-values
      valuesKey: values.yaml
